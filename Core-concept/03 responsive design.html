Customizing your theme You can completely customize your breakpoints in your tailwind.config.js file:
tailwind.config.js
/** @type {import('tailwindcss').Config} */
  module.exports = {
    theme: {
      screens: {
        'tablet': '640px',
        // => @media (min-width: 640px) { ... }
  
        'laptop': '1024px',
        // => @media (min-width: 1024px) { ... }
  
        'desktop': '1280px',
        // => @media (min-width: 1280px) { ... }
      },
    }
  }
